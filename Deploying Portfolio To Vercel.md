javascript// Toutes les images projet utilisent ce pattern
// L'image est invisible jusqu'au scroll, puis apparaît proprement

const projectImages = document.querySelectorAll('.project-img');

const imgObserver = new IntersectionObserver((entries) => {
  entries.forEach(entry => {
    if (entry.isIntersecting) {
      const img = entry.target;
      // Charger l'image réelle
      img.src = img.dataset.src;
      img.onload = () => {
        // Révéler avec GSAP une fois chargée
        gsap.to(img, {
          opacity: 1,
          y: 0,
          duration: 0.7,
          ease: 'power3.out'
        });
      };
      imgObserver.unobserve(img);
    }
  });
}, { threshold: 0.1, rootMargin: '100px' });

projectImages.forEach(img => {
  gsap.set(img, { opacity: 0, y: 30 });
  imgObserver.observe(img);
});

js// projects-scroll.js
// ─────────────────────────────────────────────────
// À importer dans main.js APRÈS l'init Lenis

function initProjectsScroll() {

  const section    = document.querySelector('.projects-scroll');
  const track      = document.querySelector('.projects-track');
  const counterEl  = document.querySelector('.counter-current');
  const progressBar = document.querySelector('.projects-progress-bar');

  if (!section || !track) return;

  // ── Calculer la distance horizontale totale à parcourir
  const getScrollAmount = () => {
    const trackWidth    = track.scrollWidth;
    const wrapperWidth  = track.parentElement.offsetWidth;
    return -(trackWidth - wrapperWidth + 80);  // +80 = padding droit
  };

  // ── L'animation principale
  const tween = gsap.to(track, {
    x: getScrollAmount,   // Fonction recalculée au resize
    ease: 'none',         // Linéaire = lié 1:1 au scroll
    scrollTrigger: {
      trigger: section,
      start: 'top top',          // Dès que la section touche le haut
      end: () => `+=${Math.abs(getScrollAmount()) + window.innerWidth}`,
      pin: true,                  // ÉPINGLE la section pendant le scroll
      scrub: 1,                   // Lag de 1s = feeling premium
      anticipatePin: 1,
      invalidateOnRefresh: true,  // Recalcule au resize

      // ── Mise à jour du compteur et de la barre
      onUpdate: (self) => {
        const cards = document.querySelectorAll('.project-card');
        const progress = self.progress;

        // Compteur actif
        const activeIndex = Math.min(
          Math.floor(progress * cards.length) + 1,
          cards.length
        );
        if (counterEl) {
          counterEl.textContent = String(activeIndex).padStart(2, '0');
        }

        // Barre de progression
        if (progressBar) {
          progressBar.style.width = `${progress * 100}%`;
        }
      }
    }
  });

  // ── Recalcul propre au resize
  window.addEventListener('resize', () => {
    tween.scrollTrigger.refresh();
  });
}

// ── Appel
initProjectsScroll();

/**
 * 14. Projects Horizontal Scroll (Sticky + Track)
 */
function initProjectsScroll() {
  const section = document.querySelector('.projects-scroll');
  const track = document.querySelector('.projects-track');
  const counterEl = document.querySelector('.counter-current');
  const progressBar = document.querySelector('.projects-progress-bar');

  if (!section || !track) return;

  // Calculer la distance horizontale totale (Track - Wrapper)
  const getScrollAmount = () => {
    const trackWidth = track.scrollWidth;
    const wrapperWidth = track.parentElement.offsetWidth;
    return -(trackWidth - wrapperWidth + 100); 
  };

  // Animation avec ScrollTrigger Pinning
  const tween = gsap.to(track, {
    x: getScrollAmount,
    ease: 'none',
    scrollTrigger: {
      trigger: section,
      start: 'top top',
      end: () => `+=${Math.abs(getScrollAmount()) + window.innerWidth}`,
      pin: true,
      scrub: 1,
      invalidateOnRefresh: true,
      onUpdate: (self) => {
        const cards = document.querySelectorAll('.project-card');
        const progress = self.progress;

        // Update Compteur (Active Index)
        const activeIndex = Math.min(
          Math.floor(progress * cards.length) + 1,
          cards.length
        );
        if (counterEl) {
          counterEl.textContent = String(activeIndex).padStart(2, '0');
        }

        // Update Progress Bar
        if (progressBar) {
          progressBar.style.width = `${progress * 100}%`;
        }
      }
    }
  });

  // Ensure refresh on resize
  window.addEventListener('resize', () => {
    ScrollTrigger.refresh();
  });
}

function initNumbersParallax() {

/**
 * 19. Interactive CV 3D Tilt
 */
function initCVInteraction() {
    const card = document.getElementById('cv-card');
    if (!card) return;

    const handleMove = (e) => {
        const rect = card.getBoundingClientRect();
        const x = e.clientX - rect.left;
        const y = e.clientY - rect.top;
        
        const centerX = rect.width / 2;
        const centerY = rect.height / 2;
        
        const rotateX = (y - centerY) / 10;
        const rotateY = (centerX - x) / 10;
        
        gsap.to(card, {
            rotateX: rotateX,
            rotateY: rotateY,
            duration: 0.5,
            ease: 'power2.out'
        });
    };

    const handleLeave = () => {
        gsap.to(card, {
            rotateX: 0,
            rotateY: 0,
            duration: 1,
            ease: 'elastic.out(1, 0.3)'
        });
    };

    card.addEventListener('mousemove', handleMove);
    card.addEventListener('mouseleave', handleLeave);
}
